# aiohttp powers the per-profile CDP reverse proxy (cdp_proxy.py).
# Plain TCP forwarders like socat can't rewrite the HTTP Host header,
# and Chrome >=136 rejects CDP HTTP requests where Host isn't
# localhost/127.0.0.1 (DNS-rebinding defense). aiohttp gives us a
# unified HTTP+WS proxy in ~80 lines.
aiohttp>=3.10,<4
# Playwright pinned to v1.58.0 because that's the multiarch image we build FROM
# (mcr.microsoft.com/playwright/python:v1.58.0-noble is the latest stable on
# Microsoft's registry as of v0.7.0). Mismatched Python-side and image-side
# Playwright versions are a known foot-gun (browser-binary path differences);
# keep them locked together.
playwright==1.58.0

# Soft anti-bot fingerprint masking applied per-page via add_init_script:
